4. Integration Code

Example JavaScript code:

// Configure providers
FiatOnRamp.configure({
    moonpayApiKey: 'pk_test_...',
    transakApiKey: 'YOUR_API_KEY',
    provider: 'moonpay' // or 'transak' or 'simulation'
});

// Set wallet address (from your app)
ObeliskApp.state.account = '0x...';

// Register callback for payment completion
FiatOnRamp.onPaymentComplete((result) => {
    console.log('Payment completed:', result);
    // Update user balance, show notification, etc.
});

// Open modal
FiatOnRamp.showModal(100); // preset amount

// Or directly open provider widget
FiatOnRamp.openMoonPayWidget({
    currencyCode: 'usdc',
    walletAddress: '0x...',
    baseCurrencyAmount: 100,
    baseCurrencyCode: 'eur'
});
